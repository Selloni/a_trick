Дополнительная полезность, если вы разобрались.
Показывает сколько место занимают файлы, в каталогах выше, удобно для поиска тяжелых папок
# Function ds() display disk usage statistics
function ds()
{
 echo
 df -H ~ | awk '{print $2,"\t"$3,"\t"$4}'
 echo
 du -h -d 1 2>/dev/null | sort -h | awk '{l[NR] = $0} END {for (i=1; i<=NR;  i++) print l[i]}'
 echo
}
